202 std::string typeToSQLiteType(const std::type_index &typeId) const
203 {
204 if (typeId == typeid(int))
205 return "INTEGER";
206 else if (typeId == typeid(double))
207 return "REAL";
208 else if (typeId == typeid(std::string))
209 return "TEXT";

223 std::string formatValue(const FieldInfo &field) const
224 {
225 if (field.typeId == typeid(int))
226 return std::to_string(std::any_cast<std::reference_wrapper<int>>(field.value).get());
227 else if (field.typeId == typeid(double))
228 return std::to_string(std::any_cast<std::reference_wrapper<double>>(field.value).get());
229 else if (field.typeId == typeid(std::string))
230 return "'" + std::any_cast<std::reference_wrapper<std::string>>(field.value).get() + "'";
231 else if (field.typeId == typeid(bool))
232 return std::to_string(std::any_cast<std::reference_wrapper<bool>>(field.value).get() ? 1 : 0);
